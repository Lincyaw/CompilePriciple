void read_program(const char *file, string &prog) {

bool getNext(char &ch, string::size_type &pos, const string &program) {
    pos++;
    if (pos >= program.size()) {
        return false;
    } else {
        ch = program[pos];
        return true;
    }
}

pair<int, string> scanner(const string &program, string::size_type &pos) {
    pair<int, string> ret;
    string token;
    int idx = 0;
    char ch = program[pos];
    while (IS_BLANK(ch)) {
        pos++;
        ch = program[pos];
    }
    if ((ch >= 'a' && ch <= 'z') || (ch >= 'A' && ch <= 'Z') || ch == '_') {
        while ((ch >= '0' && ch <= '9') || (ch >= 'a' && ch <= 'z') || (ch >= 'A' && ch <= 'Z') || ch == '_') {
            token += ch;
            if (!getNext(ch, pos, program))
            {
                break;
            }
        }

        idx = charMap.size();


        map<string, int>::const_iterator cit = charMap.find(token);
        if (cit != charMap.end()) {
            idx = cit->second;
        }
    } else if ((ch >= '0' && ch <= '9') || ch == '.')
    {
        while (ch >= '0' && ch <= '9' || ch == '.') {
            token += ch;
            if (!getNext(ch, pos, program)) {
                break;
            }
        }
        idx = (int) charMap.size() - 1;
        int dot_num = 0;
        for (string::size_type i = 0; i != token.size(); i++) {
            if (token[i] == '.') {
                dot_num++;
            }
        }
        if (dot_num > 1) {
            idx = -1;
        }

        switch (ch) {
            case '-':
                token += ch;
                if (getNext(ch, pos, program)) {
                    if (ch == '-' || ch == '=' || ch == '>')
                    {
                        token += ch;
                        getNext(ch, pos, program);
                    }
                }
                cit = charMap.find(token);
                if (cit != charMap.end()) {
                    idx = cit->second;
                }
                break;
            case '!':
                token += ch;
                if (getNext(ch, pos, program)) {
                    if (ch == '=')
                    {
                        token += ch;
                        getNext(ch, pos, program);
                    }
                }
                cit = charMap.find(token);
                if (cit != charMap.end()) {
                    idx = cit->second;
                }
                break;
            case '%':
                token += ch;
                if (getNext(ch, pos, program)) {
                    if (ch == '=')
                    {
                        token += ch;
                        getNext(ch, pos, program);
                    }
                }
                cit = charMap.find(token);
                if (cit != charMap.end()) {
                    idx = cit->second;
                }
                break;
            case '*':

                    if (ch == '=')
                    {
                        token += ch;
                        getNext(ch, pos, program);
                    }
                }
                cit = charMap.find(token);

                    idx = cit->second;

            case '^':

                    if (ch == '=')
                    {
                        token += ch;
                        getNext(ch, pos, program);

                }

                if (cit != charMap.end()) {
                    idx = cit->second;

                break;
            case '=':

                if (getNext(ch, pos, program)) {
                    if (ch == '=')
                    {
                        token += ch;
                        getNext(ch, pos, program);
                    }
                }
                cit = charMap.find(token);
                if (cit != charMap.end()) {
                    idx = cit->second;
                }
                break;

            case '/':

                    if (ch == '=')
                    {
                        token += ch;
                        getNext(ch, pos, program);

                }
                cit = charMap.find(token);
                if (cit != charMap.end()) {
                    idx = cit->second;

            case '&':

                        token += ch;
                        getNext(ch, pos, program);
                    }
                }
                cit = charMap.find(token);
                if (cit != charMap.end()) {
                    idx = cit->second;
                }
                break;

            case '|':
                token += ch;
                if (getNext(ch, pos, program)) {
                    if (ch == '|' || ch == '=') {
                        token += ch;
                        getNext(ch, pos, program);
                    }
                }
                cit = charMap.find(token);
                if (cit != charMap.end()) {
                    idx = cit->second;

                    idx = cit->second;
                }
                break;
            case '{':
                token += ch;
                cit = charMap.find(token);
                getNext(ch, pos, program);

                if (cit != charMap.end()) {
                    idx = cit->second;
                }
                break;
            case '}':
                token += ch;
                cit = charMap.find(token);
                getNext(ch, pos, program);
                if (cit != charMap.end()) {
                    idx = cit->second;
                }
                break;
            case '~':
                token += ch;
                cit = charMap.find(token);
                getNext(ch, pos, program);
                if (cit != charMap.end()) {
                    idx = cit->second;
                }
                break;
            case '\\':
                token += ch;
                if (getNext(ch, pos, program)) {
                    token += ch;
                    getNext(ch, pos, program);
                }
                cit = charMap.find(token);
                if (cit != charMap.end()) {
                    idx = cit->second;
                }
                break;
            case '"':
                token += ch;
                getNext(ch, pos, program);
                cit = charMap.find(token);
                if (cit != charMap.end()) {
                    idx = cit->second;
                }
                break;
            case '\'':
                token += ch;
                getNext(ch, pos, program);
                cit = charMap.find(token);
                if (cit != charMap.end()) {
                    idx = cit->second;
                }
                break;
            case '#':
                token += ch;
                getNext(ch, pos, program);
                cit = charMap.find(token);
                if (cit != charMap.end()) {
                    idx = cit->second;
                }
                break;
            case '\n':
                token += "nextLine";
                pos++;
                ch = program[pos];
                idx = -2;
                break;

            default:
                token += "error";
                pos++;
                ch = program[pos];
                idx = -1;
                break;
        }
    }
    ret.first = idx;
    ret.second = token;
    return ret;
}
